#### 基于功能模块的智能选择
```typescript
// file-selector.ts
interface FileSelectionContext {
    changeDescription: string;
    modifiedFiles: string[];
    projectStructure: any;
    dependencies: Map<string, string[]>;
}

class SmartFileSelector {
    private context: FileSelectionContext;

    constructor(context: FileSelectionContext) {
        this.context = context;
    }

    async selectRelevantFiles(): Promise<string[]> {
        // 1. 分析变更描述
        const keywords = this.extractKeywords(this.context.changeDescription);

        // 2. 识别功能模块
        const modules = this.identifyModules(keywords);

        // 3. 计算文件相关性分数
        const relevanceScores = await this.calculateRelevanceScores(modules);

        // 4. 选择高相关性文件
        return this.selectFilesByRelevance(relevanceScores);
    }

    private extractKeywords(description: string): string[] {
        // 使用NLP技术提取关键词
        const commonKeywords = [
            'auth', 'authentication', 'login', 'user', 'password',
            'component', 'ui', 'interface', 'view',
            'service', 'api', 'endpoint', 'backend',
            'test', 'spec', 'unit', 'integration',
            'config', 'setting', 'environment',
            'doc', 'documentation', 'readme'
        ];

        return commonKeywords.filter(keyword =>
            description.toLowerCase().includes(keyword)
        );
    }

    private identifyModules(keywords: string[]): string[] {
        const moduleMap = {
            'auth': ['src/auth/', 'src/components/auth/', 'src/services/auth/'],
            'user': ['src/user/', 'src/components/user/', 'src/models/user/'],
            'component': ['src/components/', 'src/views/', 'src/pages/'],
            'service': ['src/services/', 'src/api/', 'src/utils/'],
            'test': ['tests/', 'spec/', '__tests__/'],
            'config': ['config/', '.env', 'settings/']
        };

        const modules = new Set<string>();
        keywords.forEach(keyword => {
            if (moduleMap[keyword]) {
                moduleMap[keyword].forEach(path => modules.add(path));
            }
        });

        return Array.from(modules);
    }

    private async calculateRelevanceScores(modules: string[]): Promise<Map<string, number>> {
        const scores = new Map<string, number>();

        for (const file of this.context.modifiedFiles) {
            let score = 0;

            // 基于模块匹配计算分数
            modules.forEach(module => {
                if (file.includes(module)) {
                    score += 10;
                }
            });

            // 基于文件类型调整分数
            if (file.includes('test') || file.includes('spec')) {
                score += 5;
            }
            if (file.includes('component') || file.includes('service')) {
                score += 3;
            }
            if (file.includes('config') || file.includes('.env')) {
                score -= 2; // 配置文件通常不是功能变更
            }

            // 基于依赖关系调整分数
            const dependencies = this.context.dependencies.get(file) || [];
            dependencies.forEach(dep => {
                if (this.context.modifiedFiles.includes(dep)) {
                    score += 2;
                }
            });

            scores.set(file, score);
        }

        return scores;
    }

    private selectFilesByRelevance(scores: Map<string, number>): string[] {
        const threshold = 5; // 相关性阈值
        return Array.from(scores.entries())
            .filter(([_, score]) => score >= threshold)
            .sort((a, b) => b[1] - a[1]) // 按分数排序
            .map(([file, _]) => file);
    }
}

// 使用示例
const fileSelector = new SmartFileSelector({
    changeDescription: "Add user authentication with login and registration functionality",
    modifiedFiles: [
        'src/components/LoginForm.js',
        'src/components/RegisterForm.js',
        'src/services/authService.js',
        'src/utils/validation.js',
        'tests/auth.test.js',
        'config/dev.env',
        'package-lock.json',
        'README.md',
        'docs/api.md'
    ],
    projectStructure: {},
    dependencies: new Map([
        ['src/components/LoginForm.js', ['src/services/authService.js', 'src/utils/validation.js']],
        ['src/components/RegisterForm.js', ['src/services/authService.js', 'src/utils/validation.js']]
    ])
});

const selectedFiles = await fileSelector.selectRelevantFiles();
console.log('推荐选择的文件:', selectedFiles);
// 输出: ['src/components/LoginForm.js', 'src/components/RegisterForm.js', 'src/services/authService.js', 'src/utils/validation.js', 'tests/auth.test.js']
```

### 2.3 上下文感知的文件选择

#### 项目结构分析
```python
# project_structure_analyzer.py
import os
import ast
import json
from typing import Dict, List, Set, Tuple
from pathlib import Path

class ProjectStructureAnalyzer:
    def __init__(self, project_root: str):
        self.project_root = Path(project_root)
        self.file_tree = {}
        self.import_graph = {}
        self.function_calls = {}

    def analyze_project(self):
        """分析项目结构"""
        self.build_file_tree()
        self.analyze_imports()
        self.analyze_function_calls()
        return self.get_project_summary()

    def build_file_tree(self):
        """构建文件树结构"""
        for root, dirs, files in os.walk(self.project_root):
            # 忽略常见的排除目录
            dirs[:] = [d for d in dirs if d not in ['.git', 'node_modules', '__pycache__', 'dist', 'build']]

            relative_root = os.path.relpath(root, self.project_root)
            self.file_tree[relative_root] = {
                'files': files,
                'subdirs': dirs,
                'path': relative_root
            }

    def analyze_imports(self):
        """分析文件间的导入关系"""
        for file_path in self.get_source_files():
            try:
                with open(file_path, 'r', encoding='utf-8') as f:
                    content = f.read()

                imports = self.extract_imports(content, file_path)
                self.import_graph[file_path] = imports

            except Exception as error:
                print(f"分析文件失败 {file_path}: {error}")

    def extract_imports(self, content: str, file_path: str) -> List[str]:
        """提取文件中的导入"""
        imports = []

        if file_path.endswith('.py'):
            # Python文件
            try:
                tree = ast.parse(content)
                for node in ast.walk(tree):
                    if isinstance(node, ast.Import):
                        for alias in node.names:
                            imports.append(alias.name)
                    elif isinstance(node, ast.ImportFrom):
                        module = node.module or ''
                        for alias in node.names:
                            imports.append(f"{module}.{alias.name}")
            except:
                pass

        elif file_path.endswith(('.js', '.jsx', '.ts', '.tsx')):
            # JavaScript/TypeScript文件
            import_patterns = [
                r'import\s+.*?\s+from\s+[\'"]([^\'"]+)[\'"]',
                r'require\([\'"]([^\'"]+)[\'"]\)',
                r'import\s*\(\s*[\'"]([^\'"]+)[\'"]\s*\)'
            ]

            for pattern in import_patterns:
                matches = re.findall(pattern, content)
                imports.extend(matches)

        return imports

    def get_related_files(self, file_path: str, max_depth: int = 2) -> Set[str]:
        """获取与指定文件相关的文件"""
        related = set()
        visited = set()

        def dfs(current_file: str, depth: int):
            if depth > max_depth or current_file in visited:
                return

            visited.add(current_file)

            # 直接导入的文件
            for import_path in self.import_graph.get(current_file, []):
                # 尝试解析导入路径
                resolved_path = self.resolve_import_path(import_path, current_file)
                if resolved_path and os.path.exists(resolved_path):
                    related.add(resolved_path)
                    dfs(resolved_path, depth + 1)

            # 导入当前文件的文件
            for other_file, imports in self.import_graph.items():
                if current_file in [self.resolve_import_path(imp, other_file) for imp in imports]:
                    related.add(other_file)
                    dfs(other_file, depth + 1)

        dfs(file_path, 0)
        return related

    def resolve_import_path(self, import_path: str, current_file: str) -> str:
        """解析导入路径为实际文件路径"""
        # 简化的路径解析逻辑
        if import_path.startswith('./') or import_path.startswith('../'):
            # 相对路径
            base_dir = os.path.dirname(current_file)
            resolved = os.path.normpath(os.path.join(base_dir, import_path))

            # 尝试常见的文件扩展名
            for ext in ['.js', '.jsx', '.ts', '.tsx', '.py']:
                if os.path.exists(resolved + ext):
                    return resolved + ext
                if os.path.exists(os.path.join(resolved, 'index' + ext)):
                    return os.path.join(resolved, 'index' + ext)

        return None

    def get_project_summary(self) -> Dict:
        """获取项目结构摘要"""
        return {
            'file_tree': self.file_tree,
            'import_graph': self.import_graph,
            'total_files': len(self.get_source_files()),
            'main_directories': self.get_main_directories(),
            'dependencies': self.get_dependency_summary()
        }

# 使用示例
analyzer = ProjectStructureAnalyzer('/path/to/project')
project_summary = analyzer.analyze_project()

# 获取特定文件的关联文件
related_files = analyzer.get_related_files('src/components/UserAuth.js')
print('关联文件:', related_files)
```

### 3.2 AI驱动的提交信息生成

#### 智能提交信息生成器
```python
# commit_message_generator.py
import re
import json
from typing import List, Dict, Any
from dataclasses import dataclass

@dataclass
class CommitContext:
    changed_files: List[str]
    diff_content: str
    branch_name: str
    issue_number: str = None
    custom_description: str = ""

class CommitMessageGenerator:
    def __init__(self):
        self.patterns = {
            'file_type_mapping': {
                'test': ['test', 'spec', '__tests__'],
                'docs': ['doc', 'readme', 'md', 'txt'],
                'config': ['config', 'env', 'setting'],
                'component': ['component', 'view', 'page'],
                'service': ['service', 'api', 'controller'],
                'util': ['util', 'helper', 'lib'],
                'style': ['css', 'scss', 'less', 'style']
            },
            'operation_patterns': {
                'add': ['add', 'create', 'new', 'init'],
                'update': ['update', 'modify', 'change', 'improve'],
                'fix': ['fix', 'repair', 'resolve', 'bug'],
                'remove': ['remove', 'delete', 'cleanup'],
                'refactor': ['refactor', 'restructure', 'reorganize'],
                'optimize': ['optimize', 'improve', 'enhance']
            }
        }

    def generate_commit_message(self, context: CommitContext) -> Dict[str, Any]:
        """生成智能提交信息"""
        analysis = self.analyze_changes(context)

        commit_type = self.determine_commit_type(analysis)
        scope = self.determine_scope(analysis)
        description = self.generate_description(analysis, context)
        body = self.generate_body(analysis, context)

        return {
            'type': commit_type,
            'scope': scope,
            'description': description,
            'body': body,
            'full_message': self.format_commit_message(commit_type, scope, description, body),
            'confidence': self.calculate_confidence(analysis)
        }

    def analyze_changes(self, context: CommitContext) -> Dict[str, Any]:
        """分析代码变更"""
        analysis = {
            'file_types': self.analyze_file_types(context.changed_files),
            'operations': self.analyze_operations(context.diff_content),
            'complexity': self.analyze_complexity(context.diff_content),
            'dependencies': self.analyze_dependencies(context.changed_files)
        }

        # 检测功能模块
        analysis['modules'] = self.detect_modules(context.changed_files)

        # 检测测试覆盖率
        analysis['test_changes'] = self.detect_test_changes(context.changed_files)

        return analysis

    def analyze_file_types(self, files: List[str]) -> Dict[str, int]:
        """分析文件类型分布"""
        type_counts = {}

        for file_path in files:
            for file_type, patterns in self.patterns['file_type_mapping'].items():
                if any(pattern in file_path.lower() for pattern in patterns):
                    type_counts[file_type] = type_counts.get(file_type, 0) + 1
                    break

        return type_counts

    def analyze_operations(self, diff_content: str) -> Dict[str, int]:
        """分析操作类型"""
        operations = {'add': 0, 'remove': 0, 'modify': 0}

        # 统计添加的行
        add_lines = len(re.findall(r'^\+', diff_content.split('\n')))
        # 统计删除的行
        remove_lines = len(re.findall(r'^-', diff_content.split('\n')))

        if add_lines > remove_lines * 2:
            operations['add'] = add_lines
        elif remove_lines > add_lines * 2:
            operations['remove'] = remove_lines
        else:
            operations['modify'] = max(add_lines, remove_lines)

        return operations

    def determine_commit_type(self, analysis: Dict[str, Any]) -> str:
        """确定提交类型"""
        file_types = analysis['file_types']
        operations = analysis['operations']
        modules = analysis.get('modules', [])

        # 测试相关
        if file_types.get('test', 0) > 0 and len(file_types) == 1:
            return 'test'

        # 文档更新
        if file_types.get('docs', 0) > 0 and len(file_types) == 1:
            return 'docs'

        # 性能优化
        if 'performance' in modules or 'optimization' in modules:
            return 'perf'

        # 重构
        if 'refactor' in modules or operations.get('modify', 0) > 100:
            return 'refactor'

        # Bug修复
        if 'fix' in modules or 'bug' in modules:
            return 'fix'

        # 新功能
        if operations.get('add', 0) > operations.get('remove', 0):
            return 'feat'

        # 样式调整
        if file_types.get('style', 0) > 0:
            return 'style'

        # 默认为chore
        return 'chore'

    def determine_scope(self, analysis: Dict[str, Any]) -> str:
        """确定提交范围"""
        modules = analysis.get('modules', [])
        file_types = analysis['file_types']

        # 如果检测到模块，使用模块作为scope
        if modules:
            return modules[0]

        # 基于文件类型确定scope
        if file_types.get('component', 0) > 0:
            return 'ui'
        elif file_types.get('service', 0) > 0:
            return 'api'
        elif file_types.get('config', 0) > 0:
            return 'config'
        elif file_types.get('util', 0) > 0:
            return 'utils'

        return 'misc'

    def generate_description(self, analysis: Dict[str, Any], context: CommitContext) -> str:
        """生成提交描述"""
        # 如果用户提供了自定义描述，优先使用
        if context.custom_description:
            return context.custom_description.strip()

        operations = analysis['operations']
        modules = analysis.get('modules', [])

        # 生成描述
        if operations.get('add', 0) > 0:
            action = "添加"
        elif operations.get('remove', 0) > 0:
            action = "移除"
        else:
            action = "更新"

        if modules:
            object_desc = f"{modules[0]}功能"
        else:
            object_desc = "相关功能"

        return f"{action}{object_desc}"

    def generate_body(self, analysis: Dict[str, Any], context: CommitContext) -> str:
        """生成提交正文"""
        body_parts = []

        # 变更内容
        file_types = analysis['file_types']
        if file_types:
            changed_types = [f"{count}个{file_type}文件" for file_type, count in file_types.items()]
            body_parts.append(f"变更内容: {', '.join(changed_types)}")

        # 复杂度说明
        complexity = analysis.get('complexity', {})
        if complexity.get('high', False):
            body_parts.append("复杂度: 高 - 涉及多处修改")

        # Issue引用
        if context.issue_number:
            body_parts.append(f"关联Issue: #{context.issue_number}")

        # 测试状态
        if analysis.get('test_changes', {}).get('added', 0) > 0:
            body_parts.append("测试: 已添加相关测试用例")

        return "\n\n".join(body_parts) if body_parts else ""

    def format_commit_message(self, commit_type: str, scope: str, description: str, body: str) -> str:
        """格式化完整提交信息"""
        header = f"{commit_type}({scope}): {description}"

        if body:
            return f"{header}\n\n{body}"

        return header

    def calculate_confidence(self, analysis: Dict[str, Any]) -> float:
        """计算生成结果的置信度"""
        confidence = 0.5  # 基础置信度
```

```typescript
    path: string;
    changes: GitChange[];
    dependencies: string[];
    relevance_score: number;
    module: string;
}

interface GitChange {
    type: 'added' | 'modified' | 'deleted';
    lines_added: number;
    lines_removed: number;
    content_diff: string;
}

class ClaudeCodeGitAnalyzer {
    private projectContext: ProjectContext;
    private dependencyGraph: DependencyGraph;

    constructor(projectContext: ProjectContext) {
        this.projectContext = projectContext;
        this.dependencyGraph = new DependencyGraph();
    }

    async analyzeGitStatus(): Promise<GitFileAnalysis[]> {
        // 1. 获取Git状态
        const gitStatus = await this.getGitStatus();

        // 2. 分析每个文件的变更
        const fileAnalyses: GitFileAnalysis[] = [];

        for (const statusFile of gitStatus.modified_files) {
            const analysis = await this.analyzeFile(statusFile);
            fileAnalyses.push(analysis);
        }

        // 3. 计算文件间的关联性
        await this.calculateFileRelevance(fileAnalyses);

        return fileAnalyses;
    }

    private async analyzeFile(filePath: string): Promise<GitFileAnalysis> {
        // 获取文件变更详情
        const changes = await this.getFileChanges(filePath);

        // 分析文件模块归属
        const module = this.identifyModule(filePath);

        // 获取文件依赖关系
        const dependencies = await this.dependencyGraph.getDependencies(filePath);

        return {
            path: filePath,
            changes,
            dependencies,
            relevance_score: 0, // 将在后续计算
            module
        };
    }

    private async calculateFileRelevance(analyses: GitFileAnalysis[]): Promise<void> {
        // 构建变更摘要
        const changeSummary = this.generateChangeSummary(analyses);

        // 使用AI分析变更的主要功能
        const functionalTheme = await this.analyzeFunctionalTheme(changeSummary);

        // 计算每个文件与功能主题的相关性
        for (const analysis of analyses) {
            analysis.relevance_score = await this.calculateRelevanceScore(
                analysis,
                functionalTheme
            );
        }
    }

    private async calculateRelevanceScore(
        analysis: GitFileAnalysis,
        functionalTheme: string
    ): Promise<number> {
        let score = 0;

        // 基于文件路径匹配
        if (this.pathContainsTheme(analysis.path, functionalTheme)) {
            score += 0.4;
        }

        // 基于模块匹配
        if (analysis.module === functionalTheme) {
            score += 0.3;
        }

        // 基于变更内容分析
        const contentRelevance = await this.analyzeContentRelevance(
            analysis.changes,
            functionalTheme
        );
        score += contentRelevance * 0.3;

        return score;
    }

    async selectRelevantFiles(analyses: GitFileAnalysis[]): Promise<string[]> {
        // 按相关性分数排序
        const sortedAnalyses = analyses.sort((a, b) => b.relevance_score - a.relevance_score);

        // 找到相关性分数的阈值
        const threshold = this.calculateRelevanceThreshold(sortedAnalyses);

        // 选择超过阈值的文件
        return sortedAnalyses
            .filter(analysis => analysis.relevance_score >= threshold)
            .map(analysis => analysis.path);
    }

    async generateCommitMessage(
        selectedAnalyses: GitFileAnalysis[],
        userDescription?: string
    ): Promise<string> {
        // 生成变更摘要
        const changeSummary = this.generateDetailedSummary(selectedAnalyses);

        // 如果用户提供了描述，结合使用
        if (userDescription) {
            return await this.generateMessageWithUserInput(changeSummary, userDescription);
        }

        // 完全自动生成
        return await this.generateAutomaticMessage(changeSummary);
    }
}
```

```python
    def create_branch(self, branch_info: BranchInfo) -> bool:

            # 更新主分支
            subprocess.run(['git', 'fetch', 'origin'], check=True)
            subprocess.run(['git', 'checkout', branch_info.base_branch], check=True)
            subprocess.run(['git', 'pull', 'origin', branch_info.base_branch], check=True)

            # 创建新分支
            subprocess.run(['git', 'checkout', '-b', branch_info.name], check=True)

            # 推送到远程
            subprocess.run(['git', 'push', '-u', 'origin', branch_info.name], check=True)

            # 记录分支信息
            self.log_branch_creation(branch_info)

            return True

        except subprocess.CalledProcessError as error:
            print(f"创建分支失败: {error}")
            return False

        """生成标准化分支名称"""
        # 清理描述文本
        clean_description = self.clean_description(description)

        if branch_type in ['feature', 'bugfix'] and issue_id:
            return f"{branch_type}/{issue_id}-{clean_description}"
        elif branch_type == 'hotfix':
            return f"hotfix/{clean_description}"
        elif branch_type == 'release':
            return f"release/v{description}"  # description应该是版本号
        elif branch_type == 'experimental':
            return f"experimental/{clean_description}"
        else:
            return f"{branch_type}/{clean_description}"

        # 移除特殊字符

        # 替换空格为连字符
        clean = re.sub(r'\s+', '-', clean)

        # 转换为小写
        clean = clean.lower()

        # 限制长度
        if len(clean) > 50:
            clean = clean[:50].rstrip('-')

        return clean

    def validate_branch_name(self, branch_name: str) -> bool:
        """验证分支名称格式"""
        for branch_type, pattern in self.branch_patterns.items():
            if re.match(pattern, branch_name):

        return False

    def get_branch_type(self, branch_name: str) -> Optional[str]:
```

```python
            elif file_path.endswith(('.js', '.jsx', '.ts', '.tsx')):
                issues.extend(await self.review_javascript_file(file_path, content))
            elif file_path.endswith(('.java', '.cpp', '.c')):
                issues.extend(await self.review_compiled_language_file(file_path, content))

        except Exception as error:
            issues.append(CodeIssue(
                severity='medium',
                category='logic',
                file_path=file_path,
                line_number=0,
                message=f"无法解析文件: {error}",
                suggestion="检查文件格式和编码"
            ))

        return issues

    async def review_python_file(self, file_path: str, content: str) -> List[CodeIssue]:
        """审查Python文件"""
        issues = []

        try:
            # 解析AST
            tree = ast.parse(content)

            # 检查安全问题
            issues.extend(self.check_python_security(file_path, tree))

            # 检查性能问题
            issues.extend(self.check_python_performance(file_path, tree))

            # 检查代码风格
            issues.extend(self.check_python_style(file_path, content))

            # 检查逻辑问题
            issues.extend(self.check_python_logic(file_path, tree))

        except SyntaxError as error:
            issues.append(CodeIssue(
                severity='critical',
                category='logic',
                file_path=file_path,
                line_number=error.lineno or 0,
                message=f"语法错误: {error.msg}",
                suggestion="修复语法错误"
            ))

        return issues

    def check_python_security(self, file_path: str, tree: ast.AST) -> List[CodeIssue]:
        """检查Python安全问题"""
        issues = []

        # 危险函数使用检查
        dangerous_functions = {
            'eval': '使用eval()存在安全风险',
            'exec': '使用exec()存在安全风险',
            'input': 'input()可能在Python 2中不安全',
            'pickle.loads': 'pickle.loads可能存在代码注入风险',
            'subprocess.call': 'subprocess.call可能存在命令注入风险',
            'os.system': 'os.system可能存在命令注入风险'
        }

        for node in ast.walk(tree):
            if isinstance(node, ast.Call):
                if isinstance(node.func, ast.Name):
                    func_name = node.func.id
                    if func_name in dangerous_functions:
                        issues.append(CodeIssue(
                            severity='high',
                            category='security',
                            file_path=file_path,
                            line_number=node.lineno,
                            message=dangerous_functions[func_name],
                            suggestion=f"考虑使用更安全的替代方案，避免使用{func_name}()"
                        ))

                # 检查SQL注入风险
                elif isinstance(node.func, ast.Attribute):
                    if (node.func.attr == 'execute' and
                        isinstance(node.func.value, ast.Name) and
                        node.func.value.id == 'cursor'):
                        # 检查是否使用了字符串格式化
                        if len(node.args) > 0 and self.has_string_formatting(node.args[0]):
                            issues.append(CodeIssue(
                                severity='critical',
                                category='security',
                                file_path=file_path,
                                line_number=node.lineno,
                                message="可能存在SQL注入风险",
                                suggestion="使用参数化查询而不是字符串格式化"
                            ))

        return issues

    def check_python_performance(self, file_path: str, tree: ast.AST) -> List[CodeIssue]:
        """检查Python性能问题"""
        issues = []

        for node in ast.walk(tree):
            # 检查循环中的字符串拼接
            if isinstance(node, ast.For):
                if self.check_string_concatenation_in_loop(node):
                    issues.append(CodeIssue(
                        severity='medium',
                        category='performance',
                        file_path=file_path,
                        line_number=node.lineno,
                        message="循环中进行字符串拼接性能较差",
                        suggestion="使用列表收集字符串，最后用join()拼接"
                    ))

            # 检查不必要的全局变量查找
            if isinstance(node, ast.FunctionDef):
                global_vars = self.find_global_variable_usage(node)
                if global_vars:
                    issues.append(CodeIssue(
                        severity='low',
                        category='performance',
                        file_path=file_path,
                        line_number=node.lineno,
                        message=f"使用了全局变量: {', '.join(global_vars)}",
                        suggestion="考虑将全局变量作为参数传递"
                    ))

        return issues

    async def generate_review_summary(self, issues: List[CodeIssue]) -> str:
        """生成代码审查摘要"""
        if not issues:
            return "✅ 代码审查通过，未发现问题！"

        # 按严重程度统计
        severity_counts = {}
        category_counts = {}

        for issue in issues:
            severity_counts[issue.severity] = severity_counts.get(issue.severity, 0) + 1
            category_counts[issue.category] = category_counts.get(issue.category, 0) + 1

        # 生成摘要
        summary = f"📊 代码审查摘要\n\n"
        summary += f"发现问题总数: {len(issues)}\n\n"

        summary += "🚨 严重程度分布:\n"
        for severity in ['critical', 'high', 'medium', 'low']:
            count = severity_counts.get(severity, 0)
            if count > 0:
                emoji = {'critical': '🔴', 'high': '🟠', 'medium': '🟡', 'low': '🟢'}[severity]
                summary += f"  {emoji} {severity}: {count}\n"

        summary += "\n📋 问题类别分布:\n"
        for category, count in category_counts.items():
            summary += f"  {category}: {count}\n"

        summary += "\n🎯 优先处理建议:\n"
        critical_issues = [i for i in issues if i.severity == 'critical']
        if critical_issues:
            summary += "1. 立即修复所有🔴critical级别问题\n"

        high_issues = [i for i in issues if i.severity == 'high']
        if high_issues:
            summary += "2. 尽快修复🟠high级别问题\n"

        return summary

    def get_severity_priority(self, severity: str) -> int:
        """获取严重程度优先级"""
        priority_map = {
            'critical': 4,
            'high': 3,
            'medium': 2,
            'low': 1
        }
        return priority_map.get(severity, 0)
```

```python
class ConflictResolver:
    def __init__(self):
        self.resolution_strategies = {
            'auto_resolve': self.auto_resolve_conflicts,
            'interactive_resolve': self.interactive_resolve,
            'ai_assisted': self.ai_assisted_resolve
        }

    async def resolve_merge_conflicts(self, strategy: str = 'ai_assisted') -> bool:
        """解决合并冲突"""
        try:
            # 检查是否有冲突
            conflicts = await self.detect_conflicts()

            if not conflicts:
                print("没有发现冲突")
                return True

            print(f"发现 {len(conflicts)} 个冲突文件")

            # 选择解决策略
            resolver = self.resolution_strategies.get(strategy, self.ai_assisted_resolve)
            return await resolver(conflicts)

        except Exception as error:
            print(f"冲突解决失败: {error}")
            return False

    async def detect_conflicts(self) -> List[str]:
        """检测冲突文件"""
        result = subprocess.run(['git', 'diff', '--name-only', '--diff-filter=U'],
                              capture_output=True, text=True)

        if result.returncode != 0:
            return []

        return result.stdout.strip().split('\n')

    async def auto_resolve_conflicts(self, conflicts: List[str]) -> bool:
        """自动解决冲突"""
        resolved_count = 0

        for conflict_file in conflicts:
            resolution = await self.auto_resolve_file(conflict_file)

            if resolution:
                await self.stage_file(conflict_file)
                resolved_count += 1
                print(f"✅ 自动解决冲突: {conflict_file}")
            else:
                print(f"❌ 无法自动解决冲突: {conflict_file}")

        if resolved_count > 0:
            await self.commit_resolution(f"Auto-resolve {resolved_count} conflicts")

        return resolved_count == len(conflicts)

    async def auto_resolve_file(self, file_path: str) -> bool:
        """自动解决单个文件的冲突"""
        try:
            # 读取冲突文件内容
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()

            # 解析冲突标记
            conflicts = self.parse_conflicts(content)

            if not conflicts:
                return True  # 没有冲突

            # 尝试自动解决每个冲突
            resolved_content = content
            for conflict in conflicts:
                resolution = await self.resolve_single_conflict(conflict, file_path)
                if resolution:
                    resolved_content = resolved_content.replace(
                        conflict['full_conflict'],
                        resolution
                    )
                else:
                    return False  # 无法解决某个冲突

            # 写入解决后的内容
            with open(file_path, 'w', encoding='utf-8') as f:
                f.write(resolved_content)

            return True

        except Exception as error:
            print(f"解决文件冲突失败 {file_path}: {error}")
            return False

    def parse_conflicts(self, content: str) -> List[Dict[str, Any]]:
        """解析冲突标记"""
        conflicts = []
        lines = content.split('\n')

        i = 0
        while i < len(lines):
            if lines[i].startswith('<<<<<<<'):
                conflict = {
                    'start_line': i,
                    'full_conflict': '',
                    'ours': [],
                    'theirs': [],
                    'marker': ''
                }

                # 解析冲突部分
                in_ours = False
                in_theirs = False

                while i < len(lines):
                    line = lines[i]
                    conflict['full_conflict'] += line + '\n'

                    if line.startswith('<<<<<<<'):
                        in_ours = True
                        conflict['marker'] = line[7:].strip()
                    elif line.startswith('======='):
                        in_ours = False
                        in_theirs = True
                    elif line.startswith('>>>>>>>'):
                        break
                    elif in_ours:
                        conflict['ours'].append(line)
                    elif in_theirs:
                        conflict['theirs'].append(line)

                    i += 1

                conflicts.append(conflict)

            i += 1

        return conflicts

    async def resolve_single_conflict(self, conflict: Dict[str, Any], file_path: str) -> Optional[str]:
        """解决单个冲突"""
        # 策略1: 如果内容相同，选择任意一个
        if ''.join(conflict['ours']).strip() == ''.join(conflict['theirs']).strip():
            return ''.join(conflict['ours']).strip()

        # 策略2: 基于文件类型选择
        if file_path.endswith('.json'):
            return self.resolve_json_conflict(conflict)
        elif file_path.endswith(('.yml', '.yaml')):
            return self.resolve_yaml_conflict(conflict)
        elif file_path.endswith(('.md', '.txt')):
            return self.resolve_text_conflict(conflict)

        # 策略3: 使用AI辅助决策
        return await self.ai_resolve_conflict(conflict, file_path)

    def resolve_json_conflict(self, conflict: Dict[str, Any]) -> Optional[str]:
        """解决JSON文件冲突"""
        try:
            import json

            ours_json = json.loads('\n'.join(conflict['ours']))
            theirs_json = json.loads('\n'.join(conflict['theirs']))

            # 合并JSON对象
            merged = {**theirs_json, **ours_json}
            return json.dumps(merged, indent=2, ensure_ascii=False)

        except json.JSONDecodeError:
            return None

    def resolve_yaml_conflict(self, conflict: Dict[str, Any]) -> Optional[str]:
        """解决YAML文件冲突"""
        try:
            import yaml

            ours_data = yaml.safe_load('\n'.join(conflict['ours']))
            theirs_data = yaml.safe_load('\n'.join(conflict['theirs']))

            # 简单合并策略：深度合并字典
            def deep_merge(dict1, dict2):
                result = dict1.copy()
                for key, value in dict2.items():
                    if key in result and isinstance(result[key], dict) and isinstance(value, dict):
                        result[key] = deep_merge(result[key], value)
                    else:
                        result[key] = value
                return result

            merged = deep_merge(theirs_data, ours_data)
            return yaml.dump(merged, default_flow_style=False, allow_unicode=True)

        except Exception:
            return None

    def resolve_text_conflict(self, conflict: Dict[str, Any]) -> Optional[str]:
        """解决文本文件冲突"""
        # 计算文本相似度
        ours_text = '\n'.join(conflict['ours'])
        theirs_text = '\n'.join(conflict['theirs'])

        similarity = difflib.SequenceMatcher(None, ours_text, theirs_text).ratio()

        # 如果相似度很高，选择较长的版本
        if similarity > 0.8:
            if len(ours_text) > len(theirs_text):
                return ours_text
            else:
                return theirs_text

        return None
```

```python
# git_monitoring.py
import subprocess
import json
import sqlite3
from datetime import datetime, timedelta
from typing import Dict, List, Any

class GitActivityMonitor:
    def __init__(self, db_path: str = "git_activity.db"):
        self.db_path = db_path
        self.init_database()

    def init_database(self):
        """初始化数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        cursor.execute('''
            CREATE TABLE IF NOT EXISTS commits (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                hash TEXT NOT NULL,
                author TEXT NOT NULL,
                email TEXT NOT NULL,
                date TEXT NOT NULL,
                message TEXT NOT NULL,
                files_changed INTEGER,
                insertions INTEGER,
                deletions INTEGER,
                branch TEXT NOT NULL
            )
        ''')

        cursor.execute('''
            CREATE TABLE IF NOT EXISTS file_changes (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                commit_hash TEXT NOT NULL,
                file_path TEXT NOT NULL,
                change_type TEXT NOT NULL,
                lines_added INTEGER,
                lines_removed INTEGER
            )
        ''')

        conn.commit()
        conn.close()

    async def collect_activity_data(self, days_back: int = 30) -> Dict[str, Any]:
        """收集Git活动数据"""
        end_date = datetime.now()
        start_date = end_date - timedelta(days=days_back)

        # 获取提交历史
        commits = await self.get_commit_history(start_date, end_date)

        # 分析活动模式
        activity_analysis = await self.analyze_activity_patterns(commits)

        # 生成报告
        report = await self.generate_activity_report(activity_analysis)

        # 存储数据
        await self.store_activity_data(commits)

        return report

    async def get_commit_history(self, start_date: datetime, end_date: datetime) -> List[Dict[str, Any]]:
        """获取提交历史"""
        try:
            # 使用git log获取提交历史
            cmd = [
                'git', 'log',
                f'--since={start_date.isoformat()}',
                f'--until={end_date.isoformat()}',
                '--pretty=format:%H|%an|%ae|%ad|%s',
                '--date=iso',
                '--numstat',
                '--branches=*'
            ]

            result = subprocess.run(cmd, capture_output=True, text=True, check=True)

            commits = []
            lines = result.stdout.split('\n')
            i = 0

            while i < len(lines):
                if lines[i] and '|' in lines[i]:
                    # 解析提交信息
                    commit_info = lines[i].split('|')
                    if len(commit_info) >= 5:
                        commit_data = {
                            'hash': commit_info[0],
                            'author': commit_info[1],
                            'email': commit_info[2],
                            'date': commit_info[3],
                            'message': commit_info[4],
                            'files_changed': 0,
                            'insertions': 0,
                            'deletions': 0
                        }

                        # 解析文件变更统计
                        i += 1
                        files_data = []
                        total_insertions = 0
                        total_deletions = 0

                        while i < len(lines) and lines[i]:
                            if '\t' in lines[i]:
                                parts = lines[i].split('\t')
                                if len(parts) >= 3:
                                    insertions = int(parts[0]) if parts[0] != '-' else 0
                                    deletions = int(parts[1]) if parts[1] != '-' else 0

                                    total_insertions += insertions
                                    total_deletions += deletions

                                    files_data.append({
                                        'file_path': parts[2],
                                        'insertions': insertions,
                                        'deletions': deletions
                                    })
                            i += 1

                        commit_data['files_changed'] = len(files_data)
                        commit_data['insertions'] = total_insertions
                        commit_data['deletions'] = total_deletions
                        commit_data['files'] = files_data

                        commits.append(commit_data)
                    else:
                        i += 1
                else:
                    i += 1

            return commits

        except subprocess.CalledProcessError as error:
            print(f"获取Git历史失败: {error}")
            return []

    async def analyze_activity_patterns(self, commits: List[Dict[str, Any]]) -> Dict[str, Any]:
        """分析活动模式"""
        if not commits:
            return {}

        # 基本统计
        total_commits = len(commits)
        unique_authors = len(set(commit['author'] for commit in commits))

        # 按作者统计
        author_stats = {}
        for commit in commits:
            author = commit['author']
            if author not in author_stats:
                author_stats[author] = {
                    'commits': 0,
                    'insertions': 0,
                    'deletions': 0,
                    'files_changed': 0
                }

            author_stats[author]['commits'] += 1
            author_stats[author]['insertions'] += commit['insertions']
            author_stats[author]['deletions'] += commit['deletions']
            author_stats[author]['files_changed'] += commit['files_changed']

        # 按时间分析
        time_analysis = self.analyze_commit_timing(commits)

        # 文件变更分析
        file_analysis = self.analyze_file_changes(commits)

        # 代码质量指标
        quality_metrics = self.calculate_quality_metrics(commits)

        return {
            'total_commits': total_commits,
            'unique_authors': unique_authors,
            'author_stats': author_stats,
            'time_analysis': time_analysis,
            'file_analysis': file_analysis,
            'quality_metrics': quality_metrics
        }

    def analyze_commit_timing(self, commits: List[Dict[str, Any]]) -> Dict[str, Any]:
        """分析提交时间模式"""
        # 按小时统计
        hourly_commits = [0] * 24
        # 按星期统计
        weekly_commits = [0] * 7
        # 按日期统计
        daily_commits = {}

        for commit in commits:
            date = datetime.fromisoformat(commit['date'])

            hourly_commits[date.hour] += 1
            weekly_commits[date.weekday()] += 1

            date_key = date.date().isoformat()
            daily_commits[date_key] = daily_commits.get(date_key, 0) + 1

        # 找出最活跃的时间
        peak_hour = hourly_commits.index(max(hourly_commits))
        peak_day = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'][weekly_commits.index(max(weekly_commits))]

        return {
            'hourly_distribution': hourly_commits,
            'weekly_distribution': weekly_commits,
            'daily_commits': daily_commits,
            'peak_hour': peak_hour,
            'peak_day': peak_day,
            'avg_commits_per_day': sum(daily_commits.values()) / len(daily_commits) if daily_commits else 0
        }

    def analyze_file_changes(self, commits: List[Dict[str, Any]]) -> Dict[str, Any]:
        """分析文件变更"""
        file_change_counts = {}
        file_types = {}

        for commit in commits:
            for file_data in commit.get('files', []):
                file_path = file_data['file_path']
                file_change_counts[file_path] = file_change_counts.get(file_path, 0) + 1

                # 统计文件类型
                file_ext = file_path.split('.')[-1] if '.' in file_path else 'no_extension'
                file_types[file_ext] = file_types.get(file_ext, 0) + 1

        # 找出变更最频繁的文件
        most_changed_files = sorted(file_change_counts.items(),
                                  key=lambda x: x[1],
                                  reverse=True)[:10]

        return {
            'most_changed_files': most_changed_files,
            'file_types': file_types,
            'total_unique_files': len(file_change_counts)
        }

    def calculate_quality_metrics(self, commits: List[Dict[str, Any]]) -> Dict[str, Any]:
        """计算代码质量指标"""
        total_insertions = sum(commit['insertions'] for commit in commits)
        total_deletions = sum(commit['deletions'] for commit in commits)
        total_files_changed = sum(commit['files_changed'] for commit in commits)

        # 平均每次提交的变更
        avg_commits_per_author = len(commits) / len(set(commit['author'] for commit in commits)) if commits else 0
        avg_files_per_commit = total_files_changed / len(commits) if commits else 0
        avg_lines_per_commit = (total_insertions + total_deletions) / len(commits) if commits else 0

        # 提交信息质量
        commit_message_lengths = [len(commit['message'].split()) for commit in commits]
        avg_message_length = sum(commit_message_lengths) / len(commit_message_lengths) if commit_message_lengths else 0

        return {
            'total_insertions': total_insertions,
            'total_deletions': total_deletions,
            'total_files_changed': total_files_changed,
            'avg_commits_per_author': avg_commits_per_author,
            'avg_files_per_commit': avg_files_per_commit,
            'avg_lines_per_commit': avg_lines_per_commit,
            'avg_message_length': avg_message_length
        }

    async def generate_activity_report(self, analysis: Dict[str, Any]) -> str:
        """生成活动报告"""
        report = f"""
# Git活动分析报告

## 📊 总体统计
- 总提交数: {analysis.get('total_commits', 0)}
- 参与作者数: {analysis.get('unique_authors', 0)}
- 平均每人提交数: {analysis.get('quality_metrics', {}).get('avg_commits_per_author', 0):.1f}

## 👥 作者贡献排行
"""

        author_stats = analysis.get('author_stats', {})
        sorted_authors = sorted(author_stats.items(),
                               key=lambda x: x[1]['commits'],
                               reverse=True)

        for i, (author, stats) in enumerate(sorted_authors[:10], 1):
            report += f"{i}. {author}: {stats['commits']} 次提交\n"

        time_analysis = analysis.get('time_analysis', {})
        report += f"""
## ⏰ 提交时间模式
- 最活跃时段: {time_analysis.get('peak_hour', 0)}:00
- 最活跃日期: {time_analysis.get('peak_day', 'Unknown')}
- 平均每日提交数: {time_analysis.get('avg_commits_per_day', 0):.1f}

## 📁 文件变更分析
- 总变更文件数: {analysis.get('file_analysis', {}).get('total_unique_files', 0)}
- 变更最频繁的文件:
"""

        most_changed = analysis.get('file_analysis', {}).get('most_changed_files', [])
        for file_path, count in most_changed[:5]:
            report += f"  - {file_path}: {count} 次\n"

        report += f"""
## 📈 代码质量指标
- 平均每次提交变更文件数: {analysis.get('quality_metrics', {}).get('avg_files_per_commit', 0):.1f}
- 平均每次提交变更行数: {analysis.get('quality_metrics', {}).get('avg_lines_per_commit', 0):.1f}
- 平均提交信息长度: {analysis.get('quality_metrics', {}).get('avg_message_length', 0):.1f} 词

## 💡 建议
"""

        # 生成建议
        if time_analysis.get('avg_commits_per_day', 0) < 1:
            report += "- 建议增加提交频率，保持代码的及时同步\n"

        if analysis.get('quality_metrics', {}).get('avg_message_length', 0) < 5:
            report += "- 建议编写更详细的提交信息，提高代码可追溯性\n"

        if analysis.get('quality_metrics', {}).get('avg_files_per_commit', 0) > 10:
            report += "- 考虑将大型提交拆分为多个小的、逻辑相关的提交\n"

        return report

    async def store_activity_data(self, commits: List[Dict[str, Any]]):
        """存储活动数据到数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        for commit in commits:
            # 存储提交信息
            cursor.execute('''
                INSERT OR REPLACE INTO commits
                (hash, author, email, date, message, files_changed, insertions, deletions, branch)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            ''', (
                commit['hash'],
                commit['author'],
                commit['email'],
                commit['date'],
                commit['message'],
                commit['files_changed'],
                commit['insertions'],
                commit['deletions'],
                'main'  # 默认分支
            ))

            # 存储文件变更信息
            for file_data in commit.get('files', []):
                cursor.execute('''
                    INSERT INTO file_changes
                    (commit_hash, file_path, change_type, lines_added, lines_removed)
                    VALUES (?, ?, ?, ?, ?)
                ''', (
                    commit['hash'],
                    file_data['file_path'],
                    'modified',  # 简化处理
                    file_data['insertions'],
                    file_data['deletions']
                ))

        conn.commit()
        conn.close()
```
