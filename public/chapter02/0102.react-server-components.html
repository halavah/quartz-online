实用示例

通过实际代码学习如何使用 Server Components

Server Component 示例
jsx
// PostList.server.jsx
import { db } from '@/lib/db'
import { formatDate } from '@/lib/utils'

// Server Component - 在服务器上运行
export default async function PostList() {
  // 直接访问数据库，无需 API
  const posts = await db.post.findMany({
    orderBy: { createdAt: 'desc' },
    take: 10
  })

  return (
    <div className="post-list">
      {posts.map(post => (
        <article key={post.id} className="post-card">
          <h2>{post.title}</h2>
          <p>{post.excerpt}</p>
          <time>{formatDate(post.createdAt)}</time>
        </article>
      ))}
    </div>
  )
}
Client Component 示例
jsx
// InteractiveButton.client.jsx
'use client'

import { useState } from 'react'

// Client Component - 在客户端运行
export default function InteractiveButton() {
  const [count, setCount] = useState(0)
  const [isLoading, setIsLoading] = useState(false)

  const handleClick = async () => {
    setIsLoading(true)

    // 调用服务器操作
    const response = await fetch('/api/increment', {
      method: 'POST'
    })

    const data = await response.json()
    setCount(data.count)
    setIsLoading(false)
  }

  return (
    <button
      onClick={handleClick}
      disabled={isLoading}
      className="interactive-btn"
    >
      {isLoading ? 'Loading...' : `Count: ${count}`}
    </button>
  )
}
混合使用示例
jsx
// UserProfile.server.jsx
import InteractiveButton from './InteractiveButton.client'
import { getUser } from '@/lib/users'

export default async function UserProfile({ userId }) {
  const user = await getUser(userId)

  return (
    <div className="profile">
      {/* Server Component 内容 */}
      <img src={user.avatar} alt={user.name} />
      <h1>{user.name}</h1>
      <p>{user.bio}</p>

      {/* 嵌入 Client Component */}
      <InteractiveButton userId={user.id} />
    </div>
  )
}

使用 Server Components 的推荐模式和技巧

✅ 推荐做法：
  • 尽可能使用 Server Components，特别是数据获取密集的组件
  • 将交互性功能下沉到最小化的 Client Components
  • 使用 Server Actions 处理数据变更
  • 利用 Suspense 处理加载状态
  • 保持组件职责单一，避免混合 Server/Client 逻辑
⚠️ 避免做法：
  • 在 Server Components 中使用事件处理器
  • 导入仅在客户端可用的库（如 swr, react-query）
  • 过度使用 Client Components 导致失去 Server Components 的优势
  • 在 Server 和 Client Components 之间传递 props 时序列化函数
Server Actions 示例
jsx
// actions.js
'use server'

import { revalidatePath } from 'next/cache'
import { db } from '@/lib/db'

export async function createPost(formData) {
  const title = formData.get('title')
  const content = formData.get('content')

  // 在服务器上创建文章
  await db.post.create({
    data: { title, content }
  })

  // 重新验证缓存
  revalidatePath('/posts')

  // 重定向到文章列表
  redirect('/posts')
}

迁移指南

如何将现有应用迁移到 Server Components

1. 识别迁移候选者

优先迁移以下组件到 Server Components：

  • 纯展示型组件
  • 数据获取密集型组件
  • 不需要用户交互的组件
  • 使用 useEffect 仅用于数据获取的组件

2. 渐进式迁移策略

采用自下而上的迁移方式：

  • 从叶子组件开始迁移
  • 保持父组件为 Client Component 直到所有子组件迁移完成
  • 使用 .server.js 和 .client.js 文件后缀明确区分

3. 测试和验证

  • 确保功能正常工作
  • 监控性能指标改善
  • 验证 JavaScript 包大小减少
